How to make a population pyramid in ggplot

library(tidyverse)
library(scales)
library(patchwork)
library(here)

How do you make a population pyramid in ggplot2? To make a simple population pyramid, the answer is quite straightforward, but, as you’ll see in this blog post, you can go deep to make something polished and elegant. This is exactly what I’ve done for the last few years as I’ve produced many population pyramids for the annual Oregon by the Numbers report. I tried out a new technique this year, making separate plots for women, men, and the age categories and combining them using the {patchwork} package. When the report comes out in September, you’ll see population pyramids that look like this:

Make a basic population pyramid

To make a basic population pyramid, I’ve brought in some data from Oregon by the Numbers. I’ve saved the data as an object

Read in data

oregon_population_pyramid_data <-
  read_csv(here("population-pyramid/oregon_population_pyramid_data.csv")) |>
  mutate(percent = if_else(gender == "Men", percent, -percent)) |>
  mutate(age = fct_inorder(age))

population_pyramid_single_gender_plot <-
  function(
  county_to_filter,
  gender_to_filter
  ) {
    max_percent <-
      oregon_population_pyramid_data |>
      filter(county == county_to_filter) |>
      slice_max(
        order_by = percent,
        n = 1
      ) |>
      pull(percent)

    if (gender_to_filter == "Men") {
      x_limits <- c(0, max_percent)
    }

    if (gender_to_filter == "Women") {
      x_limits <- c(-max_percent, 0)
    }

    oregon_population_pyramid_data |>
      filter(county == county_to_filter) |>
      filter(gender == gender_to_filter) |>
      ggplot(aes(
        x = percent,
        y = age
      )) +
      geom_col(fill = "#004f39") +
      annotate(
        geom = "label",
        x = 0.05,
        y = 17,
        label = "Men",
        fill = "#004f39",
        color = "white",
        label.size = 0,
        label.padding = unit(0.3, "lines")
      ) +
      scale_x_continuous(
        labels = function(x) label_percent(accuracy = 1)(abs(x)),
        breaks = pretty_breaks(),
        limits = x_limits
      ) +
      theme_void() +
      theme(
        axis.text.x = element_text(),
        panel.grid.major.x = element_line(color = "grey90")
      )
  }

population_pyramid_combined_plot <- function(county_to_filter) {
  women_plot <-
    population_pyramid_single_gender_plot(
      county_to_filter = county_to_filter,
      gender_to_filter = "Women"
    )

  men_plot <-
    population_pyramid_single_gender_plot(
      county_to_filter = county_to_filter,
      gender_to_filter = "Men"
    )

  women_plot +
    age_order_plot +
    men_plot +
    plot_layout(
      widths = c(3.5, 1, 3.5)
    )
}
